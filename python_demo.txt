def g_write2file(o_server):
	if os.path.exists(o_server.path_) == False:
		os.mkdir(o_server.path_)
	for k_,v_ in o_server.dict_.items():
		fo = open(o_server.path_ + k_, "wb")
		fo.write(v_);
		fo.close()
		print("##create file:%s\n "%(o_server.path_ + k_ ) )

	for k_,v_ in o_server.dict_.items():
		fo = open(o_server.path_ + "debug/" + k_, "wb")
		fo.write(v_);
		fo.close()
		print("##create file:%s\n "%(o_server.path_ + k_ ) )
